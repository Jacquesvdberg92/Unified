            <script>
                (function () {
                    const COLLAPSED_KEY = 'sidebar_collapsed';
                    const ORDER_KEY     = 'sidebar_order';
                    const isChatUser    = false;

                    /* ── collapsed state ── */
                    function getCollapsed() {
                        try { return JSON.parse(localStorage.getItem(COLLAPSED_KEY)) || []; }
                        catch { return []; }
                    }
                    function saveCollapsed(list) {
                        localStorage.setItem(COLLAPSED_KEY, JSON.stringify(list));
                    }

                    /* ── item order ── */
                    function getOrders() {
                        try { return JSON.parse(localStorage.getItem(ORDER_KEY)) || {}; }
                        catch { return {}; }
                    }
                    function saveOrders(orders) {
                        localStorage.setItem(ORDER_KEY, JSON.stringify(orders));
                    }

                    /* derive a stable key for a slide <li> from its anchor href */
                    function itemKey(li) {
                        const a = li.querySelector('a');
                        return a ? a.getAttribute('href') || a.textContent.trim() : li.textContent.trim();
                    }

                    /* restore saved order inside a category-items <ul> */
                    function restoreOrder(ul, categoryKey) {
                        const orders = getOrders();
                        const savedOrder = orders[categoryKey];
                        if (!savedOrder || !savedOrder.length) return;

                        const items = Array.from(ul.querySelectorAll(':scope > li.slide'));
                        const map   = {};
                        items.forEach(function (li) { map[itemKey(li)] = li; });

                        savedOrder.forEach(function (key) {
                            if (map[key]) ul.appendChild(map[key]);
                        });
                        /* append any items not present in saved order (newly added routes) */
                        items.forEach(function (li) {
                            if (!savedOrder.includes(itemKey(li))) ul.appendChild(li);
                        });
                    }

                    /* persist current DOM order of a category-items <ul> */
                    function persistOrder(ul, categoryKey) {
                        const orders = getOrders();
                        orders[categoryKey] = Array.from(ul.querySelectorAll(':scope > li.slide'))
                            .map(itemKey);
                        saveOrders(orders);
                    }

                    /* attach drag-and-drop to a category-items <ul> */
                    function enableDrag(ul, categoryKey) {
                        let dragged = null;

                        ul.querySelectorAll(':scope > li.slide').forEach(function (li) {
                            const isNoDrag = li.getAttribute('data-no-drag') === 'true' || li.classList.contains('no-drag');
                            li.setAttribute('draggable', isNoDrag ? 'false' : 'true');

                            li.addEventListener('dragstart', function (e) {
                                if (isNoDrag) {
                                    e.preventDefault();
                                    return;
                                }

                                dragged = li;
                                li.classList.add('drag--active');
                                e.dataTransfer.effectAllowed = 'move';
                            });

                            li.addEventListener('dragend', function () {
                                li.classList.remove('drag--active');
                                ul.querySelectorAll(':scope > li.slide').forEach(function (s) {
                                    s.classList.remove('drag--over');
                                });
                                dragged = null;
                            });

                            li.addEventListener('dragover', function (e) {
                                e.preventDefault();
                                e.dataTransfer.dropEffect = 'move';
                                if (li === dragged) return;
                                ul.querySelectorAll(':scope > li.slide').forEach(function (s) {
                                    s.classList.remove('drag--over');
                                });
                                li.classList.add('drag--over');
                            });

                            li.addEventListener('dragleave', function () {
                                li.classList.remove('drag--over');
                            });

                            li.addEventListener('drop', function (e) {
                                e.preventDefault();
                                li.classList.remove('drag--over');
                                if (!dragged || dragged === li) return;

                                /* insert dragged before or after target based on pointer position */
                                const rect   = li.getBoundingClientRect();
                                const midY   = rect.top + rect.height / 2;
                                const after  = e.clientY > midY;

                                if (after) {
                                    li.after(dragged);
                                } else {
                                    li.before(dragged);
                                }

                                persistOrder(ul, categoryKey);
                            });
                        });
                    }

                    async function refreshRecentChatBadges() {
                        const items = document.querySelectorAll('[data-chat-recent-item="true"]');
                        if (!items.length) return;

                        try {
                            const res = await fetch('/CsMessaging/RecentConversations?take=8', {
                                headers: { 'X-Requested-With': 'XMLHttpRequest' }
                            });
                            const data = await res.json();
                            if (!data?.success || !Array.isArray(data.conversations)) return;

                            const byId = new Map(data.conversations.map(c => [String(c.conversationId), c]));

                            items.forEach(function (li) {
                                const id = li.getAttribute('data-conversation-id') || '';
                                const row = byId.get(id);
                                const badge = li.querySelector('.sidebar-chat-unread');

                                const unread = row?.unreadCount || 0;
                                badge.textContent = String(unread);
                                badge.classList.toggle('d-none', unread <= 0);
                            });
                        } catch {
                            // ignore sidebar badge refresh failures
                        }
                    }

                    function wireSidebarChatRealtime() {
                        if (!isChatUser) return;
                        if (typeof window.signalR === 'undefined' || !window.signalR?.HubConnectionBuilder) return;

                        connection.on('ConversationReadUpdated', refreshRecentChatBadges);
                        connection.start().then(refreshRecentChatBadges).catch(function () {});
                    }

                    /* ── main init ── */
                    function init() {
                        const collapsed = getCollapsed();

                        document.querySelectorAll('.slide__category[data-category]').forEach(function (header) {
                            const group       = header.closest('.slide__category-group');
                            const categoryKey = header.getAttribute('data-category');
                            const ul          = group.querySelector('.category-items');

                                enableDrag(ul, categoryKey);
                            }

                            /* restore collapsed state */

                                group.classList.add('collapsed');
                            }

                            header.addEventListener('click', function () {
                                const isCollapsed = group.classList.toggle('collapsed');
                                const list = getCollapsed();
                                if (isCollapsed) {
                                    if (!list.includes(categoryKey)) list.push(categoryKey);
                                } else {
                                    const idx = list.indexOf(categoryKey);
                                    if (idx > -1) list.splice(idx, 1);
                                }
                                saveCollapsed(list);
                            });
                        });

                        wireSidebarChatRealtime();
                        refreshRecentChatBadges();
                    }

                    if (document.readyState === 'loading') {
                        document.addEventListener('DOMContentLoaded', init);
                    } else {
                        init();
                    }
